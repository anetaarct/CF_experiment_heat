Figure code

This chapter documents how the figures are generated. The code blocks are not evaluated during website rendering; they show the plotting logic. Final figures are called from saved PDF files in figures/raw_distributions/, figures/model_effects/, figures/experiment_interactions/, and figures/temperature_interactions/.

Raw-data distributions

These figures show the raw response distributions before model interpretation. For continuous responses, the box centre line is the median, the box spans the interquartile range, and pale points are individual nestlings.

Show raw-distribution code 
plot_raw_distribution <- function(response) {
  plot_data <- dat |>
    filter(!is.na(.data[[response]]))
  y_label <- trait_labels[[response]] %||% response

  if (response == "D2_MASS") {
    p <- ggplot(plot_data, aes(x = EXP.INC, y = .data[[response]], fill = EXP.INC)) +
      geom_boxplot(alpha = 0.72, outlier.alpha = 0.25) +
      geom_jitter(width = 0.12, alpha = 0.22, size = 1) +
      labs(x = "Incubation treatment", y = y_label)
  } else {
    p <- ggplot(plot_data, aes(x = EXP.INC, y = .data[[response]], fill = EXP.INC)) +
      geom_boxplot(alpha = 0.72, outlier.alpha = 0.25) +
      geom_jitter(width = 0.12, alpha = 0.22, size = 1) +
      facet_wrap(~ EXP.NEST) +
      labs(x = "Incubation treatment", y = y_label, fill = "Incubation")
  }

  p <- p +
    scale_fill_manual(values = c("#476F67", "#C98C1E", "#8FAEA3", "#6E93A2")) +
    theme_minimal(base_size = 12) +
    theme(panel.grid.minor = element_blank(), legend.position = "bottom")

  ggsave(file.path("figures", "raw_distributions", paste0("raw_", response, ".pdf")),
         p, width = 7.2, height = 4.8)
}

Model-effect figures for continuous responses

These figures use model predictions as the main signal. Dark points and lines show fitted means from the selected mixed model, vertical intervals are 95% confidence intervals, and pale jittered points show original observations in the background.

Show continuous model-effect code 
plot_continuous_model_effect <- function(response) {
  fit <- readRDS(file.path("models", paste0("best_model_", response, ".rds")))
  y_label <- trait_labels[[response]] %||% response

  if (response == "D2_MASS") {
    raw_data <- dat |>
      filter(!is.na(.data[[response]]), !is.na(EXP.INC))
    pred <- ggpredict(fit, terms = "EXP.INC")

    p <- ggplot() +
      geom_jitter(
        data = raw_data,
        aes(x = EXP.INC, y = .data[[response]]),
        width = 0.12,
        alpha = 0.16,
        size = 1,
        color = "grey35"
      ) +
      geom_pointrange(
        data = pred,
        aes(x = x, y = predicted, ymin = conf.low, ymax = conf.high),
        size = 0.62,
        linewidth = 0.55,
        color = "#1F3A3D"
      ) +
      labs(x = "Incubation treatment", y = paste("Predicted", y_label))
  } else {
    raw_data <- dat |>
      filter(!is.na(.data[[response]]), !is.na(EXP.INC), !is.na(EXP.NEST))
    pred <- ggpredict(fit, terms = c("EXP.INC", "EXP.NEST"))

    p <- ggplot() +
      geom_jitter(
        data = raw_data,
        aes(x = EXP.INC, y = .data[[response]]),
        width = 0.13,
        alpha = 0.13,
        size = 0.9,
        color = "grey35"
      ) +
      geom_line(
        data = pred,
        aes(x = x, y = predicted, color = group, group = group),
        linewidth = 0.85
      ) +
      geom_pointrange(
        data = pred,
        aes(x = x, y = predicted, ymin = conf.low, ymax = conf.high, color = group, group = group),
        size = 0.58,
        linewidth = 0.5,
        position = position_dodge(width = 0.18)
      ) +
      facet_wrap(~ EXP.NEST) +
      labs(x = "Incubation treatment", y = paste("Predicted", y_label), color = "Nestling treatment")
  }

  p <- p +
    scale_color_manual(values = c("#476F67", "#C98C1E", "#6E93A2", "#8FAEA3")) +
    theme_minimal(base_size = 12) +
    theme(panel.grid.minor = element_blank(), legend.position = "bottom")

  ggsave(file.path("figures", "model_effects", paste0("model_effect_", response, ".pdf")),
         p, width = 7.2, height = 4.8)
}

Interaction and temperature figures

The interaction figures use the same visual rule as the primary figures: model predictions are the main signal and raw data are shown as pale background points when the plot remains readable.

Show interaction-figure code pattern 
# Experimental interaction:
fit <- readRDS("models/experiment_interaction/experiment_interaction_models_D12_MASS.rds")
fit <- fit$interaction_gaussian
pred <- ggpredict(fit, terms = c("EXP.INC", "EXP.NEST"))

ggplot() +
  geom_jitter(
    data = dat |> filter(!is.na(D12_MASS)),
    aes(x = EXP.INC, y = D12_MASS),
    width = 0.13,
    alpha = 0.14,
    size = 0.9,
    color = "grey35"
  ) +
  geom_line(aes(x = x, y = predicted, color = group, group = group), data = pred) +
  geom_pointrange(aes(x = x, y = predicted, ymin = conf.low, ymax = conf.high, color = group), data = pred)

# External-temperature interaction:
fit <- readRDS("models/sensitivity_temperature_interaction/temperature_interaction_models_D12_MASS.rds")
fit <- fit$interaction_gaussian
pred <- ggpredict(fit, terms = c("incubation_temperature_sc [all]", "nestling_temperature_sc [-1,0,1]"))

ggplot() +
  geom_point(
    data = dat |> filter(!is.na(D12_MASS)),
    aes(x = incubation_temperature_sc, y = D12_MASS),
    alpha = 0.14,
    size = 0.8,
    color = "grey35"
  ) +
  geom_ribbon(aes(x = x, ymin = conf.low, ymax = conf.high, fill = group), data = pred, alpha = 0.16) +
  geom_line(aes(x = x, y = predicted, color = group), data = pred)
